#CREATE TABLES
CREATE TABLE `etfs` (`date` date NOT NULL,`product` varchar(20) NOT NULL,`open` varchar(20) NOT NULL,`high` varchar(20) NOT NULL,`low` varchar(20) NOT NULL,`close` varchar(20) NOT NULL,`backward_adjusted_close` varchar(20) NOT NULL,`forward_adjusted_close` varchar(20) NOT NULL,`volume` varchar(20) NOT NULL,`dividend` varchar(20) NOT NULL,PRIMARY KEY (`date`,`product`));
CREATE TABLE `funds` (`date` date NOT NULL,`product` varchar(20) NOT NULL,`close` varchar(20) NOT NULL,`asking_price` varchar(20) NOT NULL,`backward_adjusted_close` varchar(20) NOT NULL,`forward_adjusted_close` varchar(20) NOT NULL,`dividend` varchar(20) NOT NULL,`capital_gain` varchar(20) NOT NULL,PRIMARY KEY (`date`,`product`));
CREATE TABLE `products` (`product` varchar(20) NOT NULL,`csi_number` varchar(20),`csi_symbol` varchar(20),`name` varchar(100) NOT NULL,`table` varchar(20) NOT NULL,`type` varchar(20) NOT NULL,`asset_class` varchar(40) DEFAULT NULL,`min_tick` varchar(20) DEFAULT NULL,`min_tick_value` varchar(20) DEFAULT NULL,`conversion_factor` varchar(20) NOT NULL,`currency` varchar(20) NOT NULL,`exchange` varchar(20),PRIMARY KEY (`product`));
CREATE TABLE `currency_rates` (`currency` varchar(20) NOT NULL,`rate` varchar(20) NOT NULL,PRIMARY KEY (`currency`));
CREATE TABLE `us_futures` (`date` date NOT NULL,`product` varchar(20) NOT NULL,`specific_ticker` varchar(20) NOT NULL,`open` varchar(20) NOT NULL,`high` varchar(20) NOT NULL,`low` varchar(20) NOT NULL,`close` varchar(20) NOT NULL,`is_last_trading_day` varchar(5) NOT NULL, `contract_volume` varchar(20) NOT NULL,`contract_oi` varchar(20) NOT NULL,`total_volume` varchar(20) NOT NULL,`total_oi` varchar(20) NOT NULL,PRIMARY KEY (`date`,`product`));
CREATE TABLE `euro_futures` (`date` date NOT NULL,`product` varchar(20) NOT NULL,`specific_ticker` varchar(20) NOT NULL,`open` varchar(20) NOT NULL,`high` varchar(20) NOT NULL,`low` varchar(20) NOT NULL,`close` varchar(20) NOT NULL,`is_last_trading_day` varchar(5) NOT NULL,`contract_volume` varchar(20) NOT NULL,`contract_oi` varchar(20) NOT NULL,`total_volume` varchar(20) NOT NULL,`total_oi` varchar(20) NOT NULL,PRIMARY KEY (`date`,`product`));
CREATE TABLE `uk_futures` (`date` date NOT NULL,`product` varchar(20) NOT NULL,`specific_ticker` varchar(20) NOT NULL,`open` varchar(20) NOT NULL,`high` varchar(20) NOT NULL,`low` varchar(20) NOT NULL,`close` varchar(20) NOT NULL,`is_last_trading_day` varchar(5) NOT NULL, `contract_volume` varchar(20) NOT NULL,`contract_oi` varchar(20) NOT NULL,`total_volume` varchar(20) NOT NULL,`total_oi` varchar(20) NOT NULL,PRIMARY KEY (`date`,`product`));
CREATE TABLE `other_futures` (`date` date NOT NULL,`product` varchar(20) NOT NULL,`specific_ticker` varchar(20) NOT NULL,`open` varchar(20) NOT NULL,`high` varchar(20) NOT NULL,`low` varchar(20) NOT NULL,`close` varchar(20) NOT NULL,`is_last_trading_day` varchar(5) NOT NULL, `contract_volume` varchar(20) NOT NULL,`contract_oi` varchar(20) NOT NULL,`total_volume` varchar(20) NOT NULL,`total_oi` varchar(20) NOT NULL,PRIMARY KEY (`date`,`product`));
CREATE TABLE `forex` (`date` date NOT NULL,`product` varchar(20) NOT NULL,`open` varchar(20) NOT NULL,`high` varchar(20) NOT NULL,`low` varchar(20) NOT NULL,`close` varchar(20) NOT NULL,PRIMARY KEY (`date`,`product`));
CREATE TABLE benchmarks (date date NOT NULL,benchmark varchar(20) NOT NULL,returns varchar(20) NOT NULL, PRIMARY KEY (date,benchmark));
CREATE TABLE distributions (product varchar(20) NOT NULL, declaration_date date NOT NULL, ex_dividend_date date NOT NULL, record_date date NOT NULL, payment_date date NOT NULL, amount varchar(20) NOT NULL, frequency varchar(20) NOT NULL, category varchar(20) NOT NULL, distribution_type varchar(20) NOT NULL, PRIMARY KEY (ex_dividend_date,product,category,distribution_type), KEY distributionsIndex (ex_dividend_date,product,category,distribution_type));
CREATE TABLE `yield_rates` (`date` date NOT NULL,`product` varchar(20) NOT NULL,`rate` varchar(20) NOT NULL, PRIMARY KEY (`date`,`product`));

#CREATE INDEXES
CREATE INDEX etfsIndex ON etfs (`date`, `product`);
CREATE INDEX fundsIndex ON funds (`date`, `product`);
CREATE INDEX productsIndex ON products (`product`);
CREATE INDEX us_futuresIndex ON us_futures (`date`, `product`);
CREATE INDEX euro_futuresIndex ON euro_futures (`date`, `product`);
CREATE INDEX uk_futuresIndex ON uk_futures (`date`, `product`);
CREATE INDEX other_futuresIndex ON other_futures (`date`, `product`);
CREATE INDEX forexIndex ON forex (`date`, `product`);
CREATE INDEX yield_ratesIndex ON yield_rates (`date`, `product`);

# Live trading important information
CREATE TABLE `actual_orders` (
  `id` int NOT NULL AUTO_INCREMENT,
  `date` date NOT NULL,
  `product` varchar(20) NOT NULL,
  `place_time` time NOT NULL,
  `fill_time` time NOT NULL,
  `place_amount` int(11) NOT NULL,
  `fill_amount` int(11) NOT NULL,
  `order_price` varchar(20) NOT NULL,
  `fill_price` varchar(20) DEFAULT NULL,
  `commission` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`date`,`product`,`place_time`)
)

CREATE TABLE `broker_orders` (
  `date` date NOT NULL,
  `product` varchar(20) NOT NULL,
  `amount` varchar(20) NOT NULL,
  `trade_price` varchar(20) NOT NULL,
  `commission` varchar(20) NOT NULL
)

CREATE TABLE `broker_portfolio_stats` (
  `date` date NOT NULL,
  `converted_total_bal` varchar(20) DEFAULT NULL,
  `converted_total_ote` varchar(20) DEFAULT NULL,
  `segregated_USD_bal` varchar(20) DEFAULT NULL,
  `segregated_USD_ote` varchar(20) DEFAULT NULL,
  `secured_USD_bal` varchar(20) DEFAULT NULL,
  `secured_USD_ote` varchar(20) DEFAULT NULL,
  `secured_EUR_bal` varchar(20) DEFAULT NULL,
  `secured_EUR_ote` varchar(20) DEFAULT NULL,
  `secured_GBP_bal` varchar(20) DEFAULT NULL,
  `secured_GBP_ote` varchar(20) DEFAULT NULL,
  `portfolio_value` varchar(20) DEFAULT NULL,
  `pnl` varchar(20) DEFAULT NULL,
  `maintainence_margin` varchar(20) DEFAULT NULL,
  `initial_margin` varchar(20) DEFAULT NULL,
  `commission` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`date`)
)

CREATE TABLE `commission_rates` (
  `product` varchar(20) NOT NULL DEFAULT '',
  `currency` varchar(20) DEFAULT NULL,
  `exchange_fee` varchar(20) DEFAULT NULL,
  `broker_fee` varchar(20) DEFAULT NULL,
  `clearing_fee` varchar(20) DEFAULT NULL,
  `nfa_fee` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`product`)
)

CREATE TABLE `currency_rates` (
  `date` date NOT NULL,
  `currency` varchar(20) NOT NULL,
  `rate` varchar(20) NOT NULL,
  PRIMARY KEY (`date`,`currency`)
)

CREATE TABLE `estimated_portfolio_stats` (
  `date` date NOT NULL,
  `converted_total_bal` varchar(20) DEFAULT NULL,
  `converted_total_ote` varchar(20) DEFAULT NULL,
  `segregated_USD_bal` varchar(20) DEFAULT NULL,
  `segregated_USD_ote` varchar(20) DEFAULT NULL,
  `secured_USD_bal` varchar(20) DEFAULT NULL,
  `secured_USD_ote` varchar(20) DEFAULT NULL,
  `secured_EUR_bal` varchar(20) DEFAULT NULL,
  `secured_EUR_ote` varchar(20) DEFAULT NULL,
  `secured_GBP_bal` varchar(20) DEFAULT NULL,
  `secured_GBP_ote` varchar(20) DEFAULT NULL,
  `portfolio_value` varchar(20) DEFAULT NULL,
  `pnl` varchar(20) DEFAULT NULL,
  `commission` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`date`)
)

CREATE TABLE `positions` (
  `date` date NOT NULL,
  `product` varchar(20) NOT NULL DEFAULT '',
  `estimated_position` int(11) DEFAULT NULL,
  `broker_position` int(11) DEFAULT NULL,
  `estimated_average_trade_price` varchar(20) DEFAULT NULL,
  `broker_average_trade_price` varchar(20) DEFAULT NULL,
  `broker_close_price` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`date`,`product`)
)

CREATE TABLE `inventory` (
  `date` date NOT NULL,
  `product` varchar(20) NOT NULL,
  `net_position` int(11) DEFAULT NULL,
  `strategy_position` VARCHAR(20) DEFAULT NULL,
